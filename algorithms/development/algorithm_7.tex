\begin{center}
    % Top Rule

    \vspace{0.3em}
    % Caption
    \captionof{algorithm}{Smart-Greedy Rollout Probe with Tabu List}
    \label{alg:smart_greedy_rollout_probe}
    \vspace{0.2em}
    % Middle Rule
    \hrule height 0.5pt
\end{center}

\begin{algorithmic}[1] % [1] enables line numbering
    
    % INPUT
    \Input
    \begin{itemize}
        \item $L_{\text{start}}$. The starting label $(v, t, C, \text{pred}, \text{start\_time})$
        \item $E_s$. Set of scheduled edges $(v, u, \text{dep}, \text{arr})$
        \item $E_t$. Set of transfer edges $(v, w, \Delta)$
        \item $\text{country}(v)$. Country associated with node $v$
    \end{itemize}

    % OUTPUT
    \Output
    \begin{itemize}
        \item $L_{\text{final}}$. The final label reached when the 24-hour clock expires
    \end{itemize}
    \vspace{0.5em}

    % INITIALIZATION
    \State $L \leftarrow L_{\text{start}}$
    \State $\text{RolloutVisitedNodes} \leftarrow \{L.\text{node}\}$ \Comment{Tabu list to prevent all local loops}

    \vspace{0.5em}
    \CommentLine{Repeat until no more valid moves within 24 hours}
    \Repeat
        \State $BestNodeArrivals \leftarrow$ Empty Map \Comment{Maps a node 'u' to the earliest arrival time 'tnew' and departure 'dep'}

        \vspace{0.5em}
        \CommentLine{1. Collect ONLY the earliest arrival for each neighboring node}
        \ForAll{$(L.\text{node}, u, \text{dep}, \text{arr}) \in E_s$}
            \If{$\text{dep} \ge L.t \And \text{arr} - L.\text{start\_time} \le 1440$}
                \If{$u \notin BestNodeArrivals \Or \text{arr} < BestNodeArrivals[u].\text{arr}$}
                    \State $BestNodeArrivals[u] \leftarrow (\text{dep}, \text{arr}, \text{country}(u))$
                \EndIf
            \EndIf
        \EndFor

        \ForAll{$(L.\text{node}, w, \Delta) \in E_t$}
            \State $t_{\text{new}} \leftarrow L.t + \Delta$
            \If{$t_{\text{new}} - L.\text{start\_time} \le 1440$}
                \If{$w \notin BestNodeArrivals \Or t_{\text{new}} < BestNodeArrivals[w].\text{arr}$}
                    \State \Comment{For transfers, dep and arr are both tnew}
                    \State $BestNodeArrivals[w] \leftarrow (t_{\text{new}}, t_{\text{new}}, \text{country}(w))$
                \EndIf
            \EndIf
        \EndFor

        \vspace{0.5em}
        \If{$BestNodeArrivals$ is empty}
            \State \textbf{break} \Comment{No valid forward moves left}
        \EndIf

        \vspace{0.5em}
        \CommentLine{2. Assign heavily biased weights to the deduplicated list}
        \State $ValidSteps \leftarrow \emptyset$
        \State $\text{curr\_country} \leftarrow \text{country}(L.\text{node})$

        \ForAll{$u \in BestNodeArrivals$}
            \State $(\text{dep}, \text{arr}, c_{\text{new}}) \leftarrow BestNodeArrivals[u]$
            \State $\Delta t \leftarrow \max(1, \text{arr} - L.t)$ \Comment{Total time consumed by this step}

            \If{$c_{\text{new}} \notin L.C$}
                \State \Comment{UNVISITED COUNTRY: Absolute highest priority.}
                \State $\text{base\_weight} \leftarrow 100,000$
            \ElsIf{$c_{\text{new}} \neq \text{curr\_country}$}
                \State \Comment{TRANSIT VIA OTHER COUNTRY: Crucial for escaping geographical traps.}
                \State $\text{base\_weight} \leftarrow 1,000$
            \Else
                \State \Comment{DOMESTIC TRANSIT: Just used to get to hubs.}
                \State $\text{base\_weight} \leftarrow 1$
            \EndIf
            \State $\text{weight} \leftarrow \text{base\_weight} / \Delta t$ \Comment{Linear time penalty}
            \State \Comment{THE TABU PENALTY: Prevent wandering in local loops}
            \If{$u \in \text{RolloutVisitedNodes}$}
                \State $\text{weight} \leftarrow \text{weight} \times 0.0001$
            \EndIf

            \State $ValidSteps \leftarrow ValidSteps \cup \{(u, \text{arr}, c_{\text{new}}, \text{weight})\}$
        \EndFor

        \vspace{0.5em}
        \CommentLine{3. Roll the weighted dice}
        \State $s_{\text{chosen}} \leftarrow \text{WeightedRandomChoice}(ValidSteps)$

        \vspace{0.5em}
        \CommentLine{4. Update State}
        \State $C' \leftarrow L.C \cup \{s_{\text{chosen}}.c_{\text{new}}\}$
        \State $\text{RolloutVisitedNodes} \leftarrow \text{RolloutVisitedNodes} \cup \{s_{\text{chosen}}.u\}$
        \State $L \leftarrow (s_{\text{chosen}}.u, s_{\text{chosen}}.\text{arr}, C', \text{pred}=L, \text{start\_time}=L.\text{start\_time})$

    \Until{$\text{false}$}

    \State \Return $L$

\end{algorithmic}
