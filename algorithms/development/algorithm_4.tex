In this initial version, we use the following scoring formula: 

$Score(L) = |C| + (1 - \tau)UB$; where $\tau = \frac{t - \text{start\_time}}{24:00}$
\vspace{0.5em}

\begin{algorithmic}[1] % [1] enables line numbering
    
    % INPUT
    \Input
    \begin{itemize}
        \item $V$. Set of nodes
        \item $E_s$. Set of scheduled edges $(v,u,dep,arr)$
        \item $E_t$. Set of transfer edges $(v,w,\Delta)$
        \item $\text{country}(v)$. Country associated with node $v$
        \item $R(v,b)$. Precomputed reachable country sets
        \item $\text{bucket}(t)$. Time bucket mapping function
        \item $\text{record}$. Current record to beat (e.g. 13)
    \end{itemize}

    % OUTPUT
    \Output
    \begin{itemize}
        \item $P$. Sequence of labels representing a route within 24 hours
    \end{itemize}
    \vspace{0.5em} 

    % INITIALIZATION
    \State Initialize $\text{Labels}(v) \leftarrow \emptyset$ for all $v \in V$
    \State Initialize priority queue $Q \leftarrow \emptyset$ \Comment{ordered by decreasing $score, |C|, UB, -t$}
    \State Initialize $\text{best\_label} \leftarrow \text{null}$

    \vspace{0.5em}
    \CommentLine{Initialize labels at departures}
    \ForAll{$v \in V$}
        \ForAll{scheduled departure from $v$ at time $dep$ with arrival $arr$}
            \State $L \leftarrow (v,dep,\{\text{country}(v)\},\text{pred}=\text{null},\text{start\_time}=dep)$
            \State $\text{Labels}(v) \leftarrow \text{Labels}(v) \cup \{L\}$
            \State compute $\text{score}(L)$ 
            \State $Q \leftarrow Q \cup \{L\}$
        \EndFor
    \EndFor

    \vspace{0.5em}
    \CommentLine{Main loop}
    \While{$Q \neq \emptyset$}
        \State extract $L=(v,t,C,\text{pred},\text{start\_time})$ with maximum score from $Q$

        \CommentLine{Enforce 24-hour journey limit}
        \If{$t - \text{start\_time} > 24:00$}
            \State \textbf{continue}
        \EndIf

        \CommentLine{Upper bound computation}
        \State $b \leftarrow \text{bucket}(t)$
        \State $\text{UB} \leftarrow |R(v,b) \setminus C|$

        \CommentLine{Record-based pruning}
        \If{$|C| + \text{UB} \le \text{record}$}
            \State \textbf{continue}
        \EndIf

        \CommentLine{Update best label}
        \If{$\text{best\_label}=\text{null} \lor |C|>|\text{best\_label}.C|$}
            \State $\text{best\_label} \leftarrow L$
            \If{$|C| > \text{record}$}
                \State \textbf{break} \Comment{early stopping: record beaten}
            \EndIf
        \EndIf

        \vspace{0.5em}
        \CommentLine{Extend via scheduled edges}
        \ForAll{$(v,u,dep,arr)\in E_s$}
            \If{$dep \ge t \land dep - \text{start\_time} \le 24:00$}
                \State $t_{\text{new}} \leftarrow arr$
                \If{$t_{\text{new}} - \text{start\_time} > 24:00$}
                    \State \textbf{continue}
                \EndIf
                \State $C' \leftarrow C \cup \{\text{country}(u)\}$
                \State $L' \leftarrow (u,t_{\text{new}},C',\text{pred}=L,\text{start\_time}=\text{start\_time})$
                
                \Statex % Empty line for better readability of the multiline condition
                \If{$\nexists L''\in\text{Labels}(u)$ such that \par
                    \hskip\algorithmicindent $(L''.t \le t_{\text{new}} \land L''.C \supseteq C' \land 
                    (L''.t < t_{\text{new}} \lor L''.C \neq C'))$}
                    
                    \State remove all $L''\in\text{Labels}(u)$ such that \par
                    \hskip\algorithmicindent $(t_{\text{new}} \le L''.t \land C' \supseteq L''.C \land 
                    (t_{\text{new}} < L''.t \lor C' \neq L''.C))$
                    \State $\text{Labels}(u) \leftarrow \text{Labels}(u) \cup \{L'\}$
                    \State compute $\text{score}(L')$
                    \State $Q \leftarrow Q \cup \{L'\}$
                \EndIf
            \EndIf
        \EndFor

        \vspace{0.5em}
        \CommentLine{Extend via transfers}
        \ForAll{$(v,w,\Delta)\in E_t$}
            \State $t_{\text{new}} \leftarrow t + \Delta$
            \If{$t_{\text{new}} - \text{start\_time} > 24:00$}
                \State \textbf{continue}
            \EndIf
            \State $L' \leftarrow (w,t_{\text{new}},C,\text{pred}=L,\text{start\_time}=\text{start\_time})$
            
            \Statex % Empty line
            \If{$\nexists L''\in\text{Labels}(w)$ such that \par
                \hskip\algorithmicindent $(L''.t \le t_{\text{new}} \land L''.C \supseteq C \land 
                (L''.t < t_{\text{new}} \lor L''.C \neq C))$}
                
                \State remove all $L''\in\text{Labels}(w)$ such that \par
                \hskip\algorithmicindent $(t_{\text{new}} \le L''.t \land C \supseteq L''.C \land 
                (t_{\text{new}} < L''.t \lor C \neq L''.C))$
                \State $\text{Labels}(w) \leftarrow \text{Labels}(w) \cup \{L'\}$
                \State compute $\text{score}(L')$
                \State $Q \leftarrow Q \cup \{L'\}$
            \EndIf
        \EndFor
    \EndWhile

    \vspace{0.5em}
    \CommentLine{Reconstruct path}
    \State $P \leftarrow$ empty sequence
    \State $L \leftarrow \text{best\_label}$
    \While{$L \neq \text{null}$}
        \State prepend $L$ to $P$
        \State $L \leftarrow L.\text{pred}$
    \EndWhile
    \State \Return $P$

\end{algorithmic}

% Bottom Rule
\hrule height 0.8pt

\vspace{1.5em}
In Algorithm 4 we make a change that targets one of the main issues in Algorithm 2: the explosion of labels, which are rarely pruned due to the very relaxed conditions of dominance. 

As seen in the notes, we now prioritize labels based on a scoring function that takes into account the number of countries visited, the Upper Bound of reachable countries from each node at a specific time (bucket) and the time of the label. This way we hope to expand more promising nodes first.

This is combined with an early stopping condition that breaks the loop as soon as we beat the record.Expanding promising labels earlier, combined with the early stopping, should result in a faster algorithm.

We also add a pruning condition such that labels where the sum of current number of countries and upper bound of reachable countries is less than the record are discarded. This also saves us time, since we don't care about solutions that are guaranteed to be worse than the record.
