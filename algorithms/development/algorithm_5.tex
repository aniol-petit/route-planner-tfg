    % Caption
    \captionof{algorithm}{Minimum Transit Time (MTT) Precomputation}
    \label{alg:mtt_precomputation}
    \vspace{0.2em}
    % Middle Rule
    \hrule height 0.5pt
\end{center}

\begin{algorithmic}[1] % [1] enables line numbering
    
    % INPUT
    \Input
    \begin{itemize}
        \item $V$. Set of nodes
        \item $E_s$. Set of scheduled edges $(v,u,dep,arr)$
        \item $E_t$. Set of transfer edges $(v,w,\Delta)$
        \item $\text{country}(v)$. Country associated with node $v$
    \end{itemize}

    % OUTPUT
    \Output
    \begin{itemize}
        \item $MTT$. Map of each country to the absolute minimum time required to enter it
    \end{itemize}
    \vspace{0.5em} 

    \State Initialize $MTT(c) \leftarrow \infty$ for all $c \in$ unique countries

    \vspace{0.5em}
    \CommentLine{Process scheduled edges}
    \ForAll{$(v,u,dep,arr) \in E_s$}
        \If{$\text{country}(v) \neq \text{country}(u)$}
            \State $\text{transit\_time} \leftarrow arr - dep$
            \If{$\text{transit\_time} < MTT(\text{country}(u))$}
                \State $MTT(\text{country}(u)) \leftarrow \text{transit\_time}$
            \EndIf
        \EndIf
    \EndFor

    \vspace{0.5em}
    \CommentLine{Process transfer edges (e.g., cross-border footpaths or local transit)}
    \ForAll{$(v,w,\Delta) \in E_t$}
        \If{$\text{country}(v) \neq \text{country}(w)$}
            \If{$\Delta < MTT(\text{country}(w))$}
                \State $MTT(\text{country}(w)) \leftarrow \Delta$
            \EndIf
        \EndIf
    \EndFor

    \State \Return $MTT$

\end{algorithmic}
