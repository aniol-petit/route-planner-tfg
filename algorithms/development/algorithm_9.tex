\setcounter{algorithm}{8}
\begin{center}
    % Top Rule
    \hrule height 0.8pt
    \vspace{0.3em}
    % Caption
    \captionof{algorithm}{Label-setting with Global Hard Floor, Time-Budgeted UB, and Monte-Carlo Rollouts}
    \label{alg:label_setting_global_hard_floor_time_budgeted_ub_monte_carlo_rollouts}
    \vspace{0.2em}
    % Middle Rule
    \hrule height 0.5pt
\end{center}

\begin{algorithmic}[1] % [1] enables line numbering
    
    % INPUT
    \Input
    \begin{itemize}
        \item $V$. Set of nodes
        \item $E_s$. Set of scheduled edges $(v,u,\text{dep},\text{arr})$
        \item $E_t$. Set of transfer edges $(v,w,\Delta)$
        \item $\text{country}(v)$. Country associated with node $v$
        \item $R(v,b)$. Precomputed reachable country sets
        \item $MTT(c)$. Precomputed minimum transit times for each country
        \item $\text{bucket}(t)$. Time bucket mapping function
        \item $\text{record}$. Current record to beat (e.g., 13). Static.
        \item $\text{rollout\_freq}$. Number of iterations between rollouts (e.g., 1000)
        \item $\text{GlobalMinTime}$. Minimum realistic time to visit a country (e.g., 30 mins)
    \end{itemize}

    % OUTPUT
    \Output
    \begin{itemize}
        \item $P$. Sequence of labels representing a route within 24 hours
    \end{itemize}
    \vspace{0.5em} 

    % INITIALIZATION
    \State Initialize $\text{Labels}(v) \leftarrow \emptyset$ for all $v \in V$
    \State Initialize priority queue $Q \leftarrow \emptyset$ \Comment{ordered by decreasing score}
    \State Initialize best label $\leftarrow \text{null}$
    \State $\text{iters} \leftarrow 0$ \Comment{Iteration counter for rollouts}

    \vspace{0.5em}
    \CommentLine{Initialize labels at departures}
    \ForAll{$v \in V$}
        \ForAll{scheduled edges $(v,u,\text{dep},\text{arr}) \in E_s$}
            \State $L \leftarrow (u,\text{arr},\{\text{country}(u)\},\text{pred}=\text{null},\text{start\_time}=\text{dep})$
            \State $\text{Labels}(u) \leftarrow \text{Labels}(u) \cup \{L\}$
            \State compute $\text{score}(L)$
            \State $Q \leftarrow Q \cup \{L\}$
        \EndFor
    \EndFor

    \vspace{0.5em}
    \CommentLine{Main loop}
    \While{$Q \neq \emptyset$}
        \State extract $L=(v,t,C,\text{pred},\text{start\_time})$ with maximum score from $Q$

        \vspace{0.5em}
        \CommentLine{--- GLOBAL HARD FLOOR PRUNING ---}
        \State $\text{needed} \leftarrow (\text{record} + 1) - |C|$
        \State $\text{remaining\_time} \leftarrow 1440 - (t - \text{start\_time})$
        \CommentLine{If we need countries but lack the physical time (e.g., < 30 mins per country), kill it.}
        \If{$\text{needed} > 0 \land \text{remaining\_time} < (\text{needed} \times \text{GlobalMinTime})$}
            \State \textbf{continue}
        \EndIf
        \CommentLine{---------------------------------}

        \vspace{0.5em}
        \CommentLine{--- MONTE CARLO ROLLOUT TRIGGER ---}
        \State $\text{iters} \leftarrow \text{iters} + 1$
        \If{$\text{iters} \bmod \text{rollout\_freq} = 0$}
            \CommentLine{Unconditional rollout for any label that survived Hard Floor}
            \State $L_{\text{rollout}} \leftarrow \text{Algorithm 7}(L,E_s,E_t,\text{country})$
            \If{best label $= \text{null} \lor |L_{\text{rollout}}.C| > |\text{best label}.C|$}
                \State best label $\leftarrow L_{\text{rollout}}$
            \EndIf
            \CommentLine{Early stopping if rollout beats the static record}
            \If{$|\text{best label}.C| > \text{record}$}
                \State \textbf{break}
            \EndIf
        \EndIf
        \CommentLine{----------------------------------}

        \vspace{0.5em}
        \CommentLine{Enforce 24-hour journey limit (1440 absolute minutes)}
        \If{$\text{remaining\_time} < 0$}
            \State \textbf{continue}
        \EndIf

        \vspace{0.5em}
        \CommentLine{Upper bound computation (Time-Budgeted)}
        \State $b \leftarrow \text{bucket}(t)$
        \State $U \leftarrow R(v,b) \setminus C$
        \State Sort $U$ as a sequence $(c_1, c_2, \ldots, c_k)$ such that $MTT(c_1) \le MTT(c_2) \le \ldots \le MTT(c_k)$
        \State $\text{time\_spent} \leftarrow 0$
        \State $\text{UB} \leftarrow 0$
        \For{$i = 1$ to $|U|$}
            \State $\text{time\_spent} \leftarrow \text{time\_spent} + MTT(c_i)$
            \If{$\text{time\_spent} \le \text{remaining\_time}$}
                \State $\text{UB} \leftarrow \text{UB} + 1$
            \Else
                \State \textbf{break}
            \EndIf
        \EndFor

        \vspace{0.5em}
        \CommentLine{Record-based pruning (Using static record)}
        \If{$|C| + \text{UB} \le \text{record}$}
            \State \textbf{continue}
        \EndIf

        \vspace{0.5em}
        \CommentLine{Update best label from normal queue expansion}
        \If{best label $= \text{null} \lor |C| > |\text{best label}.C|$}
            \State best label $\leftarrow L$
        \EndIf

        \CommentLine{Early stopping if normal expansion beats the static record}
        \If{$|C| > \text{record}$}
            \State \textbf{break}
        \EndIf

        \vspace{0.5em}
        \CommentLine{Extend via scheduled edges}
        \ForAll{$(v,u,\text{dep},\text{arr})\in E_s$}
            \If{$\text{dep} \ge t \And \text{dep} - \text{start\_time} \le 1440$}
                \State $t_{\text{new}} \leftarrow \text{arr}$
                \If{$t_{\text{new}} - \text{start\_time} > 1440$}
                    \State \textbf{continue}
                \EndIf
                \State $C' \leftarrow C \cup \{\text{country}(u)\}$
                \State $L' \leftarrow (u,t_{\text{new}},C',\text{pred}=L,\text{start\_time}=\text{start\_time})$
                
                \Statex % Empty line for better readability of the multiline condition
                \If{$\nexists L''\in\text{Labels}(u)$ such that \par
                    \hskip\algorithmicindent $(L''.t \le t_{\text{new}} \land L''.C \supseteq C' \land 
                    (L''.t < t_{\text{new}} \lor L''.C \neq C'))$}
                    
                    \State remove all $L''\in\text{Labels}(u)$ such that \par
                    \hskip\algorithmicindent $(t_{\text{new}} \le L''.t \land C' \supseteq L''.C \land 
                    (t_{\text{new}} < L''.t \lor C' \neq L''.C))$
                    \State $\text{Labels}(u) \leftarrow \text{Labels}(u) \cup \{L'\}$
                    \State compute $\text{score}(L')$
                    \State $Q \leftarrow Q \cup \{L'\}$
                \EndIf
            \EndIf
        \EndFor

        \vspace{0.5em}
        \CommentLine{Extend via transfers}
        \ForAll{$(v,w,\Delta)\in E_t$}
            \State $t_{\text{new}} \leftarrow t + \Delta$
            \If{$t_{\text{new}} - \text{start\_time} > 1440$}
                \State \textbf{continue}
            \EndIf
            \State $L' \leftarrow (w,t_{\text{new}},C,\text{pred}=L,\text{start\_time}=\text{start\_time})$
            
            \Statex % Empty line
            \If{$\nexists L''\in\text{Labels}(w)$ such that \par
                \hskip\algorithmicindent $(L''.t \le t_{\text{new}} \land L''.C \supseteq C \land 
                (L''.t < t_{\text{new}} \lor L''.C \neq C))$}
                
                \State remove all $L''\in\text{Labels}(w)$ such that \par
                \hskip\algorithmicindent $(t_{\text{new}} \le L''.t \land C \supseteq L''.C \land 
                (t_{\text{new}} < L''.t \lor C \neq L''.C))$
                \State $\text{Labels}(w) \leftarrow \text{Labels}(w) \cup \{L'\}$
                \State compute $\text{score}(L')$
                \State $Q \leftarrow Q \cup \{L'\}$
            \EndIf
        \EndFor
    \EndWhile

    \vspace{0.5em}
    \CommentLine{Reconstruct path}
    \State $P \leftarrow$ empty sequence
    \State $L \leftarrow$ best label
    \While{$L \neq \text{null}$}
        \State prepend $L$ to $P$
        \State $L \leftarrow L.\text{pred}$
    \EndWhile
    \State \Return $P$

\end{algorithmic}

\vspace{1.5em}
After running Algorithm 8 and debugging, we observed an important issue: most of labels that were object of the rollouts were very close to the time-line end, with a very low country count. For example: [ROLLOUT] iters=316,000 | pre t=1373 | pre |C|=7 | post |C|=7.

This led to the idea of adding a global hard floor pruning mechanism. This uses a minimum realistic time per country (set arbitrarily and, of course, optimistically to avoid missing complicated but feasible routes) to aggressively prune labels that cannot possibly reach the record even with optimal remaining paths. Additionally, the rollout trigger is optimized to only execute when there is a comfortable time buffer, and the record is dynamically updated during search to enable immediate pruning of the queue.

Hence, this is a very similar algorithm to Algorithm 8, but with the addition of the global hard floor pruning mechanism which we hope can drastically reduce the labels that stay in the queue and are expanded, hoping for a faster convergence.
